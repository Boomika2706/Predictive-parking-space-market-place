  <script>
    async function fetchSpaces() {
      const res = await fetch("/api/spaces");
      const data = await res.json();
      const tbody = document.querySelector("#spaceTable tbody");
      tbody.innerHTML = "";
      data.forEach(s => {
        const row = `<tr>
          <td>${s.space_id}</td>
          <td>${s.location}</td>
          <td>${s.occupied ? "Occupied" : "Free"}</td>
          <td>${(s.predicted_free_prob * 100).toFixed(0)}%</td>
          <td><button onclick="reserve('${s.space_id}')">Reserve</button></td>
        </tr>`;
        tbody.innerHTML += row;
      });
    }

    async function addSpace() {
      await fetch("/api/create_space", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ space_id: "A" + Math.floor(Math.random()*100), location: "Zone 1" })
      });
      fetchSpaces();
    }

    async function reserve(id) {
      const res = await fetch("/api/reserve", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ space_id: id, user: "demo_user", duration: 15 })
      });
      const msg = await res.json();
      alert(msg.message);
      fetchSpaces();
    }

    fetchSpaces();
    setInterval(fetchSpaces, 5000);
  </script>
